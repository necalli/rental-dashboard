import json

import subprocess

import requests

procs = []
logs = []


def sh(cmd, check=True):
    p = subprocess.run(["bash", "-lc", cmd], text=True)
    if check and p.returncode != 0:
        raise RuntimeError(f"Command failed ({p.returncode}): {cmd}")


def start_bg(name, cmd, log_path, cwd=None, env=None):
    log_file = open(log_path, "w", encoding="utf-8")
    p = subprocess.Popen(cmd, cwd=cwd, env=env, stdout=log_file, stderr=subprocess.STDOUT)
    procs.append((name, p))
    logs.append(log_file)
    return p


def wait_http(url, timeout=180):
    end = time.time() + timeout
    last = None
    while time.time() < end:
        try:
            r = requests.get(url, timeout=8)
            if r.status_code < 500:
                return r
            last = f"status={r.status_code}"
        except Exception as exc:
            last = str(exc)
        time.sleep(2)
    raise RuntimeError(f"Timeout waiting for {url}. Last error: {last}")


def tail(path, lines=80):
    try:
        return subprocess.check_output(["bash", "-lc", f"tail -n {int(lines)} {path}"], text=True)
    except Exception:
        return f"(could not read {path})"
